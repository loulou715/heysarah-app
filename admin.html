<script>

  const API = 'https://script.google.com/macros/s/AKfycby9JB0Y54FGdqVOPmWhjC83Nm8DcEW2XYXnO6Nnhx9yVWnPVcXbJCvBoo5l-oj4WDk9zA/exec';

  let allOrders = [];


  // ── 頁面切換 ─────────────────────────────────────

  function showPage(name) {

    ['dashboard','orders','import','summary'].forEach(p => {

      document.getElementById('page-' + p).classList.toggle('hidden', p !== name);

    });

    document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));

    event.currentTarget.classList.add('active');

    if (name === 'dashboard') loadDashboard();

    if (name === 'orders')    loadOrders('all');

    if (name === 'summary')   loadSummary('all');

  }


  // ── 總覽 ─────────────────────────────────────────

  async function loadDashboard() {

    const data = await apiFetch('allOrders', { status: 'all' });

    if (!data.success) return;

    const orders = data.orders;

    document.getElementById('stat-total').textContent   = orders.length;

    document.getElementById('stat-pending').textContent = orders.filter(o=>o.status==='待確認').length;

    document.getElementById('stat-paid').textContent    = orders.filter(o=>o.status==='已付款').length;

    document.getElementById('tb-recent').innerHTML = orders.slice(0,8).map(o=>`

      <tr>

        <td style="color:#999;font-size:.78rem">${o.timestamp.substring(5,16)}</td>

        <td>${o.nickname}</td>

        <td>[${o.productCode}] ${o.productName}</td>

        <td>${o.qty}</td>

        <td style="color:#b8975a">NT$${o.subtotal.toLocaleString()}</td>

        <td><span class="badge badge-${o.status}">${o.status}</span></td>

      </tr>`).join('');

  }


  // ── 訂單管理 ─────────────────────────────────────

  async function loadOrders(status) {

    const data = await apiFetch('allOrders', { status });

    if (!data.success) return;

    allOrders = data.orders;

    renderOrders(allOrders);

  }


  function filterOrders(status, el) {

    document.querySelectorAll('#page-orders .tab').forEach(t=>t.classList.remove('active'));

    el.classList.add('active');

    const filtered = status === 'all' ? allOrders : allOrders.filter(o=>o.status===status);

    renderOrders(filtered);

  }


  function renderOrders(orders) {

    document.getElementById('tb-orders').innerHTML = orders.map(o=>`

      <tr>

        <td style="font-size:.72rem;color:#aaa">${o.orderId}</td>

        <td>${o.nickname}</td>

        <td>[${o.productCode}] ${o.productName}</td>

        <td>${o.spec || '—'}</td>

        <td>${o.qty}</td>

        <td style="color:#b8975a">NT$${o.subtotal.toLocaleString()}</td>

        <td><span class="badge badge-${o.status}">${o.status}</span></td>

        <td>

          <select onchange="updateStatus('${o.orderId}', this.value)" style="font-size:.75rem;padding:3px 6px;border-radius:6px;border:1px solid #eee">

            ${['待確認','已付款','已出貨','取消'].map(s=>`<option ${s===o.status?'selected':''}>${s}</option>`).join('')}

          </select>

        </td>

      </tr>`).join('');

  }


  async function updateStatus(orderId, status) {

    await apiFetch('updateStatus', { orderId, status });

  }


  // ── 批次匯入 ─────────────────────────────────────

  async function doImport() {

    const text     = document.getElementById('importText').value.trim();

    const nickname = document.getElementById('importNickname').value.trim() || '後台匯入';

    const el       = document.getElementById('import-result');

    if (!text) { el.textContent = '請貼上訂單文字'; return; }


    // 客戶端解析（全形相容）

    const normalized = text

      .replace(/[Ａ-Ｚ]/g, c=>String.fromCharCode(c.charCodeAt(0)-0xFEE0))

      .replace(/[ａ-ｚ]/g, c=>String.fromCharCode(c.charCodeAt(0)-0xFEE0))

      .replace(/[０-９]/g, c=>String.fromCharCode(c.charCodeAt(0)-0xFEE0))

      .replace(/＋/g,'+');


    const regex = /([A-Za-z]\d+)\s*([^+\n]*?)\s*\+\s*(\d+)/gi;

    const orders = [];

    let m;

    while ((m = regex.exec(normalized)) !== null) {

      orders.push({

        productCode: m[1].toUpperCase(),

        spec: m[2].trim().replace(/^\/+|\/+$/g,'').trim(),

        qty: parseInt(m[3]),

        nickname

      });

    }


    if (!orders.length) { el.textContent = '❌ 未解析到訂單，請確認格式（如：A01黑L+1）'; return; }

    el.textContent = `解析到 ${orders.length} 筆，匯入中⋯`;


    const data = await apiFetch('import', { orders: JSON.stringify(orders) });

    el.textContent = data.success ? `✅ 成功匯入 ${data.count} 筆訂單` : '❌ ' + data.message;

  }


  // ── 備貨統計 ─────────────────────────────────────

  async function loadSummary(filter, el) {

    if (el) {

      document.querySelectorAll('#page-summary .tab').forEach(t=>t.classList.remove('active'));

      el.classList.add('active');

    }

    const data = await apiFetch('summary', { filter });

    if (!data.success) return;

    document.getElementById('tb-summary').innerHTML = data.summary.map(s=>`

      <tr>

        <td><b>${s.code}</b></td>

        <td>${s.name}</td>

        <td>${s.spec || '—'}</td>

        <td style="font-size:1.1rem;font-weight:500">${s.qty}</td>

        <td style="color:#999">${s.orders} 筆</td>

      </tr>`).join('');

    document.getElementById('summary-total').textContent =

      `共 ${data.orderCount} 筆訂單　總數量：${data.total} 件`;

  }


  // ── 共用 fetch ────────────────────────────────────

  async function apiFetch(action, params={}) {

    const qs = Object.entries({action,...params}).map(([k,v])=>`${k}=${encodeURIComponent(v)}`).join('&');

    const res = await fetch(`${API}?${qs}`);

    return res.json();

  }

  loadDashboard();
