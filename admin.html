// ── 設定 ────────────────────────────────────────────────────────────── var CONFIG = { SPREADSHEET_ID: '1Q_FI5qiQkZKM3gx8PliUmkDDQjQz41dD8VbXckF_RVI', SHEET_ORDERS: '訂單', SHEET_PRODUCTS: '商品', SHEET_MEMBERS: '會員' }; // ── doGet ────────────────────────────────────────────────────────────── function doGet(e) { var p = e.parameter || {}; var action = p.action || ''; var result; try { if (action === 'allOrders') { result = getAllOrders(p.status || 'all'); } else if (action === 'getProducts' || action === 'getProductsAdmin') { result = getProductsAdmin(); } else if (action === 'saveProduct') { result = saveProduct(JSON.parse(p.product || '{}')); } else if (action === 'saveProducts') { var plist = JSON.parse(p.products || '[]'); var saved = 0; for (var k = 0; k < plist.length; k++) { var sr = saveProduct(plist[k]); if (sr.success) saved++; } result = { success: true, saved: saved }; } else if (action === 'deleteProduct') { result = deleteProduct(p.code || ''); } else if (action === 'import' || action === 'batchImport') { result = batchImportOrders(JSON.parse(p.orders || '[]')); } else if (action === 'query' || action === 'queryOrders') { result = queryOrdersByNickname(p.nickname || ''); } else if (action === 'updateStatus') { result = updateOrderStatus(p.orderId || '', p.status || ''); } else if (action === 'summary') { result = getOrderSummary(p.filter || 'all'); } else if (action === 'deleteOrders') { result = deleteOrders(JSON.parse(p.orderIds || '[]')); } else if (action === 'setup') { setupSheets(); result = { success: true, message: 'Setup complete' }; } else { result = { success: true, message: 'HEYSARAH API v4.1', version: '4.1' }; } } catch (err) { result = { success: false, message: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── 取得所有訂單（13欄結構）──────────────────────────────────────────── function getAllOrders(statusFilter) { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_ORDERS); if (!sheet) return { success: true, orders: [] }; var data = sheet.getDataRange().getValues(); var orders = []; for (var i = 1; i < data.length; i++) { var row = data[i]; if (!row[0]) continue; var status = String(row[10] || '待確認'); if (statusFilter && statusFilter !== 'all' && status !== statusFilter) continue; orders.push({ orderId: String(row[0]), timestamp: row[1] ? Utilities.formatDate(new Date(row[1]), 'Asia/Taipei', 'yyyy-MM-dd HH:mm:ss') : '', nickname: String(row[2] || ''), userId: String(row[3] || ''), productCode: String(row[4] || ''), productName: String(row[5] || ''), spec: String(row[6] || ''), qty: Number(row[7]) || 0, unitPrice: Number(row[8]) || 0, subtotal: Number(row[9]) || 0, status: status, note: String(row[11] || ''), easyStoreId: String(row[12] || '') }); } return { success: true, orders: orders }; } // ── 取得商品目錄（含下架，管理員用）───────────────────────────────────── function getProductsAdmin() { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_PRODUCTS); if (!sheet) return { success: true, products: [] }; var data = sheet.getDataRange().getValues(); var products = []; for (var i = 1; i < data.length; i++) { if (!data[i][0]) continue; products.push({ code: String(data[i][0]).toUpperCase(), name: String(data[i][1] || ''), price: Number(data[i][2]) || 0, image: String(data[i][3] || ''), status: String(data[i][4] || '上架'), easyStoreId: String(data[i][5] || ''), stock: Number(data[i][6]) || 0, salePrice: Number(data[i][7]) || 0, note: String(data[i][8] || '') }); } return { success: true, products: products }; } // ── 商品 Map（供訂單匯入查詢價格用）───────────────────────────────────── function getProductMap() { var result = getProductsAdmin(); var map = {}; if (result.success) { result.products.forEach(function(p) { map[p.code] = p; }); } return map; } // ── 儲存單一商品（新增或更新，名稱可選）───────────────────────────────── function saveProduct(p) { if (!p.code) return { success: false, message: '缺少商品編號' }; var code = String(p.code).toUpperCase().trim(); var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_PRODUCTS); // 若工作表不存在，自動建立並加入標題列 if (!sheet) { sheet = ss.insertSheet(CONFIG.SHEET_PRODUCTS); sheet.appendRow(['商品編號','品名','定價','圖片網址','狀態','EasyStoreID','庫存','連線優惠價','備註']); } var data = sheet.getDataRange().getValues(); // 找現有列，有則更新（保留原值若新值未傳入） for (var i = 1; i < data.length; i++) { if (String(data[i][0]).toUpperCase().trim() === code) { sheet.getRange(i + 1, 1, 1, 9).setValues([[ code, p.name !== undefined ? String(p.name) : String(data[i][1] || ''), p.price !== undefined ? Number(p.price) : Number(data[i][2] || 0), p.image !== undefined ? String(p.image) : String(data[i][3] || ''), p.status !== undefined ? String(p.status) : String(data[i][4] || '上架'), p.easyStoreId !== undefined ? String(p.easyStoreId) : String(data[i][5] || ''), p.stock !== undefined ? Number(p.stock) : Number(data[i][6] || 0), p.salePrice !== undefined ? Number(p.salePrice) : Number(data[i][7] || 0), p.note !== undefined ? String(p.note) : String(data[i][8] || '') ]]); return { success: true, action: 'updated', code: code }; } } // 新列 sheet.appendRow([ code, String(p.name || ''), Number(p.price || 0), String(p.image || ''), String(p.status || '上架'), String(p.easyStoreId|| ''), Number(p.stock || 0), Number(p.salePrice || 0), String(p.note || '') ]); return { success: true, action: 'created', code: code }; } // ── 刪除商品 ────────────────────────────────────────────────────────── function deleteProduct(code) { if (!code) return { success: false, message: '缺少商品編號' }; code = String(code).toUpperCase().trim(); var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_PRODUCTS); if (!sheet) return { success: false, message: '商品工作表不存在' }; var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]).toUpperCase().trim() === code) { sheet.deleteRow(i + 1); return { success: true, deleted: code }; } } return { success: false, message: '找不到商品 ' + code }; } // ── 批次匯入訂單 ────────────────────────────────────────────────────── function batchImportOrders(orders) { if (!orders || !orders.length) return { success: false, message: '無訂單資料' }; var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_ORDERS); if (!sheet) { sheet = ss.insertSheet(CONFIG.SHEET_ORDERS); sheet.appendRow(['訂單編號','訂單時間','社群暱稱','LINE userId','商品編號','商品名稱','規格','數量','單價','小計','處理狀態','備註','EasyStore訂單ID']); } var productMap = getProductMap(); var now = new Date(); var dateStr = Utilities.formatDate(now, 'Asia/Taipei', 'yyyyMMdd'); var rows = []; var count = 0; orders.forEach(function(o) { var code = String(o.productCode || '').toUpperCase().trim(); var info = productMap[code]; // 優先使用前端傳來的 unitPrice，其次查商品目錄，最後 0 var unitPrice = Number(o.unitPrice) || (info ? (info.salePrice || info.price || 0) : 0); var qty = Number(o.qty) || 1; count++; var orderId = 'HS-' + dateStr + '-' + String(Date.now()).slice(-4) + String(count).padStart(2, '0'); rows.push([ orderId, now, String(o.nickname || '後台匯入'), String(o.userId || ''), code, info ? (info.name || String(o.productName || '')) : String(o.productName || ''), String(o.spec || ''), qty, unitPrice, unitPrice * qty, '待確認', String(o.note || ''), '' ]); }); if (rows.length > 0) { sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, 13).setValues(rows); } return { success: true, count: count }; } // ── 依暱稱查詢訂單 ──────────────────────────────────────────────────── function queryOrdersByNickname(nickname) { var result = getAllOrders('all'); if (!result.success) return result; var filtered = nickname ? result.orders.filter(function(o) { return o.nickname.indexOf(nickname) !== -1; }) : result.orders; return { success: true, orders: filtered }; } // ── 更新訂單狀態 ────────────────────────────────────────────────────── function updateOrderStatus(orderId, status) { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_ORDERS); if (!sheet) return { success: false, message: '訂單表不存在' }; var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]) === orderId) { sheet.getRange(i + 1, 11).setValue(status); return { success: true }; } } return { success: false, message: '找不到訂單 ' + orderId }; } // ── 備貨彙整 ────────────────────────────────────────────────────────── function getOrderSummary(filter) { var result = getAllOrders('all'); if (!result.success) return result; var orders = result.orders.filter(function(o) { if (filter === 'pending') return o.status === '待確認'; if (filter === 'paid') return o.status === '已付款'; return o.status !== '取消'; }); var map = {}; orders.forEach(function(o) { var key = o.productCode + '||' + (o.spec || ''); if (!map[key]) { map[key] = { code: o.productCode, name: o.productName || '', spec: o.spec || '', qty: 0, orders: 0 }; } map[key].qty += o.qty; map[key].orders += 1; }); var summary = Object.keys(map).map(function(k) { return map[k]; }).sort(function(a, b) { return a.code < b.code ? -1 : a.code > b.code ? 1 : 0; }); var totalQty = summary.reduce(function(s, x) { return s + x.qty; }, 0); return { success: true, summary: summary, total: totalQty, orderCount: orders.length }; } // ── 刪除訂單 ────────────────────────────────────────────────────────── function deleteOrders(orderIds) { if (!orderIds || !orderIds.length) return { success: false, message: '無訂單ID' }; var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_ORDERS); if (!sheet) return { success: false, message: '訂單表不存在' }; var data = sheet.getDataRange().getValues(); var toDelete = []; for (var i = 1; i < data.length; i++) { if (orderIds.indexOf(String(data[i][0])) !== -1) toDelete.push(i + 1); } for (var j = toDelete.length - 1; j >= 0; j--) { sheet.deleteRow(toDelete[j]); } return { success: true, deleted: toDelete.length }; } // ── 初始化工作表 ────────────────────────────────────────────────────── function setupSheets() { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); if (!ss.getSheetByName(CONFIG.SHEET_ORDERS)) { var os = ss.insertSheet(CONFIG.SHEET_ORDERS); os.appendRow(['訂單編號','訂單時間','社群暱稱','LINE userId','商品編號','商品名稱','規格','數量','單價','小計','處理狀態','備註','EasyStore訂單ID']); } if (!ss.getSheetByName(CONFIG.SHEET_PRODUCTS)) { var ps = ss.insertSheet(CONFIG.SHEET_PRODUCTS); ps.appendRow(['商品編號','品名','定價','圖片網址','狀態','EasyStoreID','庫存','連線優惠價','備註']); } }
